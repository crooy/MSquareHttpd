Failing test¶

Please submit a failing test on the following format:

import org.scalatest.WordSpec
import org.scalatest.matchers.MustMatchers

class Ticket001Spec extends WordSpec with MustMatchers {

  "An XXX" should {
    "do YYY" in {
      1 must be (1)
    }
  }
}
